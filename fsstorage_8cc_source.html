17 bool FSStorage::splitNameRoleVersion(const std::string& full_name, std::string* role_name, int* version) {
18  size_t dot_pos = full_name.find('.');
19 
20  // doesn't have a dot
21  if (dot_pos == std::string::npos) {
22  return false;
23  }
24  std::string prefix = full_name.substr(0, dot_pos);
25  if (role_name != nullptr) {
26  *role_name = full_name.substr(dot_pos + 1);
27  }
28 
29  try {
30  auto v = boost::lexical_cast<int>(prefix);
31  if (version != nullptr) {
32  *version = v;
33  }
34  } catch (const boost::bad_lexical_cast&) {
35  return false;
36  }
37  return true;
38 }

40 Uptane::Version FSStorage::findMaxVersion(const boost::filesystem::path& meta_directory, Uptane::Role role) {
41  int version = -1;
42  if (!boost::filesystem::exists(meta_directory)) {
43  return {};
44  }
45 
46  boost::filesystem::directory_iterator it{meta_directory};
47  for (; it != boost::filesystem::directory_iterator(); ++it) {
48  if (!boost::filesystem::is_regular_file(it->path())) {
49  continue;
50  }
51  std::string name = it->path().filename().native();
52  int file_version;
53  std::string file_role;
54  if (splitNameRoleVersion(name, &file_role, &file_version)) {
55  if (file_role == Uptane::Version().RoleFileName(role) && file_version > version) {
56  version = file_version;
57  }
58  }
59  }
60 
61  return Uptane::Version(version);
62 }

64 FSStorage::FSStorage(const StorageConfig& config, bool migration_only) : INvStorage(config) {

72  if ((st.st_mode & (S_IRGRP | S_IROTH)) != 0) {

87  boost::filesystem::path director_path = config_.uptane_metadata_path.get(config_.path) / "director";
88  // migrate from old unversioned Uptane root meta

90  for (auto repo : {Uptane::RepositoryType::Director, Uptane::RepositoryType::Images}) {
91  boost::filesystem::path& meta_dir = repo == (Uptane::RepositoryType::Director) ? director_path : images_path;

99  }
100  }
101  }
